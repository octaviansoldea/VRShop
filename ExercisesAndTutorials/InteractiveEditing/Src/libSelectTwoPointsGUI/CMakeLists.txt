CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT(libSelectTwoPointsGUI)

FIND_PACKAGE(VTK)
#IF(NOT VTK_DIR)
#  MESSAGE(FATAL_ERROR "Please set VTK_DIR.")
#ENDIF(NOT VTK_DIR)
#INCLUDE(${VTK_USE_FILE})


# Find VTK.
FIND_PACKAGE(VTK REQUIRED)
IF(VTK_USE_FILE)
  INCLUDE(${VTK_USE_FILE})
ENDIF(VTK_USE_FILE)


# use what QVTK built with
SET(QT_QMAKE_EXECUTABLE ${VTK_QT_QMAKE_EXECUTABLE} CACHE FILEPATH "")
SET(QT_MOC_EXECUTABLE ${VTK_QT_MOC_EXECUTABLE} CACHE FILEPATH "")
SET(QT_UIC_EXECUTABLE ${VTK_QT_UIC_EXECUTABLE} CACHE FILEPATH "")
SET(DESIRED_QT_VERSION ${VTK_DESIRED_QT_VERSION} CACHE FILEPATH "")
FIND_PACKAGE(Qt)
IF(QT_USE_FILE)
  INCLUDE(${QT_USE_FILE})
ELSE(QT_USE_FILE)
  SET(QT_LIBRARIES   ${QT_QT_LIBRARY})
ENDIF(QT_USE_FILE)

# Use the include path and library for Qt that is used by VTK.
INCLUDE_DIRECTORIES(
        ${QT_INCLUDE_DIR}
        ${CMAKE_CURRENT_BINARY_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${CMAKE_SOURCE_DIR}/libArrowSourceStartEnd
	    ${CMAKE_CURRENT_SOURCE_DIR}/../libSphereWidget/
	    ${CMAKE_CURRENT_BINARY_DIR}/../libSphereWidget
	    ${CMAKE_CURRENT_BINARY_DIR}/../libSelectTwoPointsGUI/
)


QT4_WRAP_UI(UI_SRCS SelectTwoPointsGUI.ui)
QT4_WRAP_CPP(MOC_SRCS SelectTwoPointsGUI.h)
SET(SRCS ${SRCS} ${MOC_SRCS}
    	#main.cpp
	AxesWidget.cpp AxesWidget.h
	SelectTwoPoints.cpp SelectTwoPoints.h
	SelectTwoPointsGUI.cpp SelectTwoPointsGUI.h
	)

ADD_LIBRARY(libSelectTwoPointsGUI ${SRCS} ${UI_SRCS})

INSTALL(TARGETS libSelectTwoPointsGUI DESTINATION lib)

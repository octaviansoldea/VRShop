PROJECT(libModel)
CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

SET(CMAKE_DEBUG_POSTFIX "D")


#start Qt
IF(Qt5_FOUND)
  INCLUDE_DIRECTORIES(
    ${Qt5Core_INCLUDE_DIRS}
	${Qt5OpenGL_INCLUDE_DIRS}
    ${Qt5Widgets_INCLUDE_DIRS}
  )
ELSE(Qt5_FOUND)
  #Assume Qt4_FOUND, if we are here
  IF(QT_USE_FILE)
    INCLUDE(${QT_USE_FILE})
  ELSE(QT_USE_FILE)
    SET(QT_LIBRARIES   ${QT_QT_LIBRARY})
  ENDIF(QT_USE_FILE)

  INCLUDE_DIRECTORIES(
       ${QT_INCLUDE_DIR}
  )
ENDIF(Qt5_FOUND)
#end Qt


INCLUDE_DIRECTORIES(
	${OPENSCENEGRAPH_INCLUDE_DIRS}
	${CMAKE_BINARY_DIR}/testing/libModel
	${CMAKE_CURRENT_SOURCE_DIR}/../libCommon_OSG
	${CMAKE_CURRENT_SOURCE_DIR}/../libOSGQTViewer
)

SET(SRCS
	VRMainWindow_GUI.h VRMainWindow_GUI.cpp
)


#MOC according to the Qt version
IF(Qt5_FOUND)
  QT5_WRAP_CPP(MOC_SRCS VRMainWindow_GUI.h)
  QT5_WRAP_UI(UI_SRCS	VRMainWindow_GUI.ui)

ELSE()
  QT4_WRAP_UI(UI_SRCS	VRMainWindow_GUI.ui)
  QT4_WRAP_CPP(MOC_SRCS VRMainWindow_GUI.h)
ENDIF()


ADD_LIBRARY(libModel
  ${MOC_SRCS} ${UI_SRCS} ${SRCS}
)

SET_PROPERTY(TARGET libModel PROPERTY FOLDER Utils)
